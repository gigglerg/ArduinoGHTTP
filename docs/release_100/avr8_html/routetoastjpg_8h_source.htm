18 public:
19  bool Handler(Shared &s) {
20  if (s.request.url!=T("/image/toast.jpg") || s.request.method!=T("get")) {
21  // Not handled
22  return false;
23  }
24 
25  // Set response maximum to be size of buffer. This isnt right but enough for this example, it should be -TCP/IP headers
26  s.response.SetMaximum(s.request.GetMaximum());
27 
28  // Set response buffer
29  s.response.SetBuffer(s.misc.response_buffer, 0, 0); // Don't use request after this...
30 
31  // Set status OK
32  s.response.status.Set(ghttp::HTTPResponseStatus::STATUS_OK);
33  s.response.Header(T("HTTP/1.1"));
34 
35  s.response.Header(T("Content-Type: image/jpeg"));
36  s.response.HeaderComplete();
37 #if defined(ARDUINO_ARCH_AVR)
38  // AVR flash accessible by different bus for SRAM. There has to be a nicer way to do this??
39  s.response.Append(reinterpret_cast<PGM_VOID_P>(const_cast<char*>(__resource_toast_jpg)), sizeof(__resource_toast_jpg));
40 #else
41  // STM32 flash accessible by same bus as SRAM
42  s.response.Append(__resource_toast_jpg, sizeof(__resource_toast_jpg));
43 #endif
44  s.response.Complete();
45 
46  Serial.print("Length: ");
47  Serial.println(s.response.GetLength(), DEC);
48 
49  return true; // We have handled
50  }
