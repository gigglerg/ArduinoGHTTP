13  * An example implemented route for protocol HTTP, POST requests covering route "/counter". You would break down your application into a number of
14  * OOD route handlers like this to simplify implementation

16 class RouteCounter : public ghttp::Route<Shared> {
17 public:
18  bool Handler(Shared &s) {
19  if (s.request.url!=T("/counter") || s.request.method!=T("post") || s.request.protocol!=T("http")) {
20  // Not handled
21  return false;
22  }
23 
24 #if GHTTP_REQUEST_PROCESS_COOKIES == 1 && GHTTP_SERVER_USE_SESSIONS == 1
25  // Use this helper to get session globals + set status
26  auto& session_vars = ghttp::HTTPHelper::StartSession<Shared, Global>(s,100);
27 
28 
29  // Incrememnt a counter added to my session instance of Global class
30  session_vars.count++;
31 
32  if (s.request.query_variable[T("clear")]) { // Like PHP isset($_GET["clear"])
33  session_vars.count=0; // Like PHP $_SESSION['count']++
34  }
35 
36 #else
37  // Set status. Default is OK, no change here
38  s.response.Header(T("HTTP/1.1"));
39 
40  // Tell our response instance to use the same buffer as request to reduce memory footprint
41  sd.response.SetMaximum(sizeof(buffer));
42  sd.response.SetBuffer(buffer, 0);
43 #endif
44 
45  // No more response headers to come
46  s.response.HeaderComplete();
47 
48  // Output response data + Complete
49  s.response+=T("<html><body><p>Session counter ");
50 #if GHTTP_REQUEST_PROCESS_COOKIES == 1 && GHTTP_SERVER_USE_SESSIONS == 1
51  s.response+=ghttp::StringHelper::FromUInt<uint8_t>(session_vars.count, s.response.GetBuffer() + s.response.GetLength());
52  // or
53  // s.response.AppendUInt<uint8_t>(session_vars.count)
54 #else
55  s.response+=T("0 (NO SESSION)");
56 #endif
57  s.response+=T("</p></body></html>");
58  s.response.Complete();
59 
60  return true; // We have handled
61  }
62 }; // class RouteCounter
