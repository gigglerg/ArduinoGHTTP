12 /**
13  * An example implemented route for HTTP GET/POST requests covering route "/login"
14  */
15 class RouteLogin : public ghttp::Route<Shared> {
16 public:
17  bool Handler(Shared &s) {
18  if (s.request.url!=T("/login") || (s.request.method!=T("get") && s.request.method!=T("post"))) {
19  // Not handled
20  return false;
21  }
22 
23  // Use this helper to get session globals + set status. NOTE: arbitrary 164 characters of request still available while response building
24  auto& session_vars = ghttp::HTTPHelper::StartSession<Shared, Global>(s, 164, "path=/", s.misc.response_buffer);
25 
26  bool failed_login = false;
27 
28  // Has user post form content?
29  if (s.request.post_variable[T("user")] && s.request.post_variable[T("pass")]) { // Like PHP isset($_POST["user"]) && isset($_POST["pass"])
30  // WARNING: These type of string comparisons are NOT case sensitive. See example ghttp_ethercard_server_authorization.ino where they are
31  if (strlen(&s.request.post_variable[T("user")]->value)>0 && s.request.post_variable[T("pass")]->value==T("donotask")) {
32  // Authenticated...
33  session_vars.logged_in = true;
34  strncpy(session_vars.username, &s.request.post_variable[T("user")]->value, sizeof(session_vars.username)-1);
35  session_vars.username[sizeof(session_vars.username)-1] = '\0';
36  }else {
37  failed_login = true;
38  }
39  }
40 
41  // If logged in, redirect to our root route
42  if (session_vars.logged_in) {
43  s.response.Header(T("HTTP/1.1"));
44  s.GenericHeader();
45  ghttp::HTTPHelper::Redirect<Shared>(s, "/");
46 
47  // We didn't handle content but did issue the redirect so indicate route handled
48  return true;
49  }
50 
51  // Set status. Default is OK, no change here
52  s.response.Header(T("HTTP/1.1"));
53  s.GenericHeader();
54  s.response.HeaderComplete(); // No more headers
55 
56  // Output response data + Complete
57  const char* content = "<html><head><link rel='shortcut icon' type='image/png' href='/favicon.ico'/></head>" \
58  "<body><form action='/login' method='post'>" \
59  "<label for='user'>Username:</label><input type='text' name='user' value='' maxlength='15' required/>" \
60  "<label for='pass'>Password:</label><input type='password' name='pass' value='' maxlength='15' required/>" \
61  "<div><input type='submit' value='submit'/></div>" \
62  "</form></div>";
63  s.response.Append(content);
64  if (failed_login) {
65  s.response.Append(T("<div style='color:red;background-color:black;'>ERROR: Login attempt failed</div>"));
66  }
67  s.response+=T("</body></html>");
68  s.response.Complete();
69 
70  return true; // We have handled
71  }
72 }; // class RouteLogin
