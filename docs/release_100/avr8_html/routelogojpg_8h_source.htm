19 /**
20  * An example implemented route for HTTP GET requests on "image/logo.jpg"
21  */
22 class RouteLogoJpg : public ghttp::Route<Shared> {
23 public:
24  bool Handler(Shared &s) {
25  if (s.request.url!=T("/image/logo.jpg") || s.request.method!=T("get")) {
26  // Not handled
27  return false;
28  }
29 
30  // Set response buffer
31  s.response.SetBuffer(s.request.GetBuffer(), 0, 0); // Don't use request after this...
32 
33  // Set status OK
34  s.response.status.Set(ghttp::HTTPResponseStatus::STATUS_OK);
35  s.response.Header(T("HTTP/1.1"));
36 
37  s.response.Header(T("Content-Type: image/jpeg"));
38  /*
39  * Here we simply alter the size of the response to include raw image data length. This will alter the headers but actual buffering
40  * isn't present due to size. So we aren't going to use the SRAM buffer for anything other than the headers.
41  */
42  uint16_t raw_image_length = static_cast<uint16_t>(sizeof(__resource_ghttp_chrome_jpg));
43 
44  char content_length[24] = "Content-Length: ";
45  ghttp::StringHelper::FromUInt<uint16_t>(raw_image_length, &content_length[16]);
46  s.response.Header(content_length);
47  s.response.HeaderComplete();
48  s.response.Complete(false); // false as we've added our own content length, ignore anything in response buffer
49 
50  /*
51  * The send will be broken down into 2, first the response headers, second the raw image data
52  */
53  ECWrap::Send(s.response.GetBuffer(), s.response.GetLength(), true, true);
54 
55  // STM32 flash accessible by same bus as SRAM
56  ECWrap::Send(reinterpret_cast<const char*>(&__resource_ghttp_chrome_jpg[0]), raw_image_length, false);
57 
58  /*
59  * Last important point. Because we've handled the request and actually sent the response we don't want the main application to do
60  * anything with response buffer so I set the length to 0. This is application specific and just how I've chosen to signal this fact
61  * but you're free to do what you like...
62  */
63  s.response.SetLength(0);
64 
65  return true; // We have handled
66  }
67 }; // class RouteLogoJpg
