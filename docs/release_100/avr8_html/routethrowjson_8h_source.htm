12 /**
13  * An example implemented route for HTTP GET requests covering route "/json/throw"
14  */
15 class RouteThrowJSON : public ghttp::Route<Shared> {
16 public:
17  bool Handler(Shared &s) {
18  // Set response maximum to be size of buffer
19  if (s.request.url!=T("/json/throw") || s.request.method!=T("post")) {
20  // Not handled
21  return false;
22  }
23 
24  uint8_t die_count = 1;
25 
26  // Like PHP isset($_POST["req"]) && $_POST["req"] == "req" && isset($_POST["die"])
27  if (s.request.post_variable[T("req")] && (*s.request.post_variable[T("req")]) == T("throw") && s.request.post_variable[T("die")]) {
28  auto *dv = s.request.post_variable[T("die")]->value.Get();
29  if (dv) {
30  uint8_t dc = dv[0] - '0';
31 
32  // Take die count from client
33  die_count = dc>0 && dc<8 ? dc : 1;
34  }
35  }
36 
37  // Set response maximum to be size of buffer - no offset as not accessing request while building response
38  s.response.SetMaximum(s.request.GetMaximum());
39 
40  // Set response buffer
41  s.response.SetBuffer(s.request.GetBuffer(), 0, 0); // Don't use request after this...
42 
43  // Set status OK
44  s.response.status.Set(ghttp::HTTPResponseStatus::STATUS_OK);
45  s.response.Header(T("HTTP/1.1"));
46  s.response.Header(T("Content-Type: application/json; charset=iso-8859-1"));
47  s.response.HeaderComplete();
48  s.response.Append("[");
49  for(uint8_t i=0; i<die_count; i++) {
50  if (i) {
51  s.response.Append(",");
52  }
53 
54  // Generate die[i] face value (dots) and roll direction/orientation (0 or 1)
55  uint8_t face_value = 1 + (rand() % 6);
56  uint8_t face_direction = rand() % 2;
57 
58  s.response.Append("{\"value\":");
59  s.response.AppendUInt<uint8_t>(face_value);
60  s.response.Append(", \"dir\":");
61  s.response.AppendUInt<uint8_t>(face_direction);
62  s.response.Append("}");
63  }
64  s.response.Append("]");
65  s.response.Complete();
66 
67  return true; // We have handled
68  }
69 }; // class RouteThrowJSON
